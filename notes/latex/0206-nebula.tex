\section{Nebula}\label{nebula}}

\begin{itemize}
\tightlist
\item
  Mesh VPN
\item
  Similar to Tailscale
\item
  Does not use Wireguard
\item
  Built using the Noise Protocol Framework (used in Wireguard)
\item
  Uses a Certificate Authority that needs to sign each Peer's
  certificate

  \begin{itemize}
  \tightlist
  \item
    Certificates contain

    \begin{itemize}
    \tightlist
    \item
      Peer's Virtual IP address
    \item
      Peer's public key
    \end{itemize}
  \end{itemize}
\end{itemize}

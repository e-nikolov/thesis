\def\exec{\immediate\write18}
\def\inputAllFiles#1{%
  \def\filename{#1/\jobname.md}
  % \exec{ls #1/*.md | xargs cat >> #1/\jobname.md}%
  \exec{for i in #1/*.md; do echo >> \filename && echo "\\begin{frame}" >> \filename && cat $i >> \filename && echo >> \filename && echo "\\end{frame}" >> \filename;done}%
  \markdownInput{#1/\jobname.md}
  \AtEndDocument{\exec{cp -f #1/\jobname.md #1/\jobname.tmp; rm -f #1/\jobname.md}}
}
